import Foundation
import UIKit

public final class SafeAppSDK: NSObject, @unchecked Sendable {
    private let environment: ProctorEnvironment
    private let proctoringId: String
    private let configuration: SafeAppConfiguration
    private let apiClient: ProctorAPIClient

    private var armed = false

    private var debuggerMonitor: DebuggerMonitor?
    private var jailbreakDetector: JailbreakDetector?
    private var clipboardMonitor: ClipboardMonitor?
    private var networkMonitor: NetworkMonitor?
    private var locationMonitor: LocationMonitor?
    private var deviceInfoMonitor: DeviceInfoMonitor?
    private var appStateMonitor: AppStateMonitor?
    private var integrityService: DeviceIntegrityService?

    public init(
        environment: ProctorEnvironment,
        proctoringId: String,
        configuration: SafeAppConfiguration = .init()
    ) {
        self.environment = environment
        self.proctoringId = proctoringId
        self.configuration = configuration
        self.apiClient = ProctorAPIClient(
            environment: environment,
            timeoutSeconds: configuration.networkConfig.requestTimeoutSeconds
        )
        super.init()
    }

    public func setSecurityCallback(_ callback: SafeAppSecurityCallback?) {
        SafeAppSecurityBus.shared.register(callback)
    }

    public func onCreate() {
        SessionLogger.shared.i("EP_SDK", "onCreate prepared=true")
    }

    public func startMonitoring() {
        guard !armed else { return }
        armed = true

        SessionLogger.shared.startSession()
        SessionLogger.shared.setProctoringId(proctoringId)
        SessionLogger.shared.it("EP_SDK", "session_started proctoringId=\(proctoringId) env=\(environment.rawValue.uppercased())")
        SessionLogger.shared.it("EP_SDK", "Monitoring STARTED file=\(SessionLogger.shared.getSessionFilePath() ?? "na")")

        if configuration.enableAppStateMonitoring {
            let monitor = AppStateMonitor()
            monitor.start()
            appStateMonitor = monitor
        }

        if configuration.enableDeviceInfoMonitoring {
            let monitor = DeviceInfoMonitor()
            monitor.logNow()
            deviceInfoMonitor = monitor
        }

        if configuration.enableScreenCaptureAwareness {
            logScreenCaptureState()
        }

        if configuration.enableDebuggerDetection {
            let monitor = DebuggerMonitor(config: configuration.debuggerConfig)
            monitor.start()
            debuggerMonitor = monitor
        }

        if configuration.enableJailbreakDetection {
            let detector = JailbreakDetector(config: configuration.jailbreakConfig)
            detector.checkNow()
            jailbreakDetector = detector
        }

        if configuration.enableClipboardMonitoring {
            clipboardMonitor = ClipboardMonitor(config: configuration.clipboardConfig)
        }

        if configuration.enableNetworkMonitoring {
            let monitor = NetworkMonitor(config: configuration.networkConfig)
            monitor.start()
            networkMonitor = monitor
        }

        if configuration.enableLocationMonitoring {
            let monitor = LocationMonitor(config: configuration.locationConfig)
            monitor.start()
            locationMonitor = monitor
        }

        let integrity = DeviceIntegrityService()
        integrity.checkAvailability()
        integrity.generateDeviceToken()
        integrityService = integrity
    }

    public func stopMonitoring(reason: String = "manual_stop") {
        guard armed else { return }
        armed = false

        appStateMonitor?.stop()
        appStateMonitor = nil

        debuggerMonitor?.stop()
        debuggerMonitor = nil

        networkMonitor?.stop()
        networkMonitor = nil

        locationMonitor?.stop()
        locationMonitor = nil

        clipboardMonitor = nil
        jailbreakDetector = nil
        deviceInfoMonitor = nil
        integrityService = nil

        SessionLogger.shared.it("EP_SDK", "Monitoring STOPPED reason=\(reason)")
        SessionLogger.shared.flush(reason: "before_upload")

        if
            let path = SessionLogger.shared.getSessionFilePath(),
            let contents = try? String(contentsOfFile: path, encoding: .utf8)
        {
            Task {
                await apiClient.sendLogs(logsJSON: contents)
            }
        }

        SessionLogger.shared.closeSession(reason: reason)
    }

    public func logClipboardCopy(_ text: String?, source: String = "app_copy") {
        clipboardMonitor?.logCopy(text: text, source: source)
    }

    public func logClipboardPaste(_ text: String?, source: String = "app_paste") {
        clipboardMonitor?.logPaste(text: text, source: source)
    }

    public func getSessionFilePath() -> String? {
        SessionLogger.shared.getSessionFilePath()
    }

    public func isSessionActive() -> Bool {
        SessionLogger.shared.isSessionActive()
    }

    private func logScreenCaptureState() {
        let isCaptured = UIScreen.main.isCaptured
        SessionLogger.shared.i("EP_CAPTURE", "initial isCaptured=\(isCaptured)")
    }
}
