import Foundation

public final class ProctorAPIClient: @unchecked Sendable {
    private let environment: ProctorEnvironment
    private let httpClient: SecureHTTPClient

    public init(
        environment: ProctorEnvironment,
        timeoutSeconds: TimeInterval = 30
    ) {
        self.environment = environment
        self.httpClient = SecureHTTPClient(timeoutSeconds: timeoutSeconds)
    }

    public func sendLogs(logsJSON: String) async {
        let endpoint = environment.baseURL.appendingPathComponent("api/Log/safe-app-log")

        do {
            let quotedJSONString = try JSONEncoder().encode(logsJSON)

            SessionLogger.shared.it("EP_PROCTOR_API", "Sending logs → url=\(endpoint.absoluteString) size=\(logsJSON.count)")

            let (_, response) = try await httpClient.postJSON(to: endpoint, body: quotedJSONString)

            if (200...299).contains(response.statusCode) {
                SessionLogger.shared.it("EP_PROCTOR_API", "Logs sent successfully → HTTP \(response.statusCode)")
            } else {
                SessionLogger.shared.wt("EP_PROCTOR_API", "Failed to send logs → HTTP \(response.statusCode)")
            }
        } catch {
            SessionLogger.shared.et("EP_PROCTOR_API", "Exception while sending logs: \(error.localizedDescription)")
        }
    }
}
