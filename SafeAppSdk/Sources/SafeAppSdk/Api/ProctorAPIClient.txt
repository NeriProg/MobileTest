import Foundation

public final class ProctorAPIClient: @unchecked Sendable {
    private let environment: ProctorEnvironment
    private let session: URLSession

    public init(
        environment: ProctorEnvironment,
        timeoutSeconds: TimeInterval = 30
    ) {
        self.environment = environment

        let config = URLSessionConfiguration.default
        config.timeoutIntervalForRequest = timeoutSeconds
        config.timeoutIntervalForResource = timeoutSeconds
        self.session = URLSession(configuration: config)
    }

    public func sendLogs(logsJSON: String) async {
        let endpoint = environment.baseURL.appendingPathComponent("api/Log/safe-app-log")

        do {
            var request = URLRequest(url: endpoint)
            request.httpMethod = "POST"
            request.setValue("application/json", forHTTPHeaderField: "Content-Type")

            let quotedJSONString = try JSONEncoder().encode(logsJSON)
            request.httpBody = quotedJSONString

            SessionLogger.shared.it("EP_PROCTOR_API", "Sending logs → url=\(endpoint.absoluteString) size=\(logsJSON.count)")

            let (_, response) = try await session.data(for: request)

            guard let httpResponse = response as? HTTPURLResponse else {
                SessionLogger.shared.wt("EP_PROCTOR_API", "Failed to send logs → non-HTTP response")
                return
            }

            if (200...299).contains(httpResponse.statusCode) {
                SessionLogger.shared.it("EP_PROCTOR_API", "Logs sent successfully → HTTP \(httpResponse.statusCode)")
            } else {
                SessionLogger.shared.wt("EP_PROCTOR_API", "Failed to send logs → HTTP \(httpResponse.statusCode)")
            }
        } catch {
            SessionLogger.shared.et("EP_PROCTOR_API", "Exception while sending logs: \(error.localizedDescription)")
        }
    }
}
